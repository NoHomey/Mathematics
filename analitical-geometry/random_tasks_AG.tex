    \\A(a_1, a_2), \; B(b_1, b_2), \; C(c_1, c_2), \; D(d_1, d_2)\\

    \\S\bigtriangleup ABC = \frac{1}{2}|\vectr{AB} \times \vectr{AC}|\\
    \\\vectr{AB}(b_1 - a_1, b_2 - a_2, 0), \; \vectr{AC}(c_1 - a_1, c_2 - a_2, 0)\\
    \\S\bigtriangleup ABC = \frac{1}{2}\begin{vmatrix}
